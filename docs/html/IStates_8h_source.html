264 static void removeNode(Network * network, const Node * node) {
265
266 // Initialized at size(), so we can use < size() to check if something changed
267 size_t to_delete = network->getIStateGroup()->size();
268 for (size_t i=0; i < network->getIStateGroup()->size(); i++)
269 {
270 auto * istate_group = network->getIStateGroup()->at(i);
271 if (istate_group->hasNode(node)) {
272 if (istate_group->getNodes()->size() == 1) {
273 if (to_delete < network->getIStateGroup()->size()) {
274 throw BNException("Two IStateGroup with the same node");
275 }
276 to_delete = i;
277 // the node should be in a unique place, so it should work ?
278 break;
279
280 } else {
281 size_t ii;
282 std::vector<const Node*>* group_nodes = istate_group->getNodes();
283 for(ii = 0; ii < group_nodes->size(); ii++) {
284 if (group_nodes->at(ii) == node) {
285 group_nodes->erase(group_nodes->begin() + (std::ptrdiff_t) ii);
286 break;
287 }
288 }
289
290 std::vector<IStateGroup::ProbaIState*>* proba_istates = istate_group->getProbaIStates();
291 for (auto * proba_istate : *proba_istates)
292 {
293 proba_istate->state_value_list->erase(proba_istate->state_value_list->begin() + (std::ptrdiff_t) i);
294 }
295 }
296 }
297 }
298 if (to_delete < network->getIStateGroup()->size())
299 network->getIStateGroup()->erase(network->getIStateGroup()->begin() + (std::ptrdiff_t) to_delete);
300 }

302 static void setStatesProbas(Network * network, std::vector<const Node*>* nodes, std::map<std::vector<bool>, double>& probas) {
303 for (auto* node: *nodes) {
304 IStateGroup::removeNode(network, node);
305 }
306
307 std::vector<IStateGroup::ProbaIState*>* new_proba_istates = new std::vector<IStateGroup::ProbaIState*>();
308 for (auto& state_proba: probas) {
309
310 std::vector<double>* state_values = new std::vector<double>();
311 for (auto node_state: state_proba.first) {
312 state_values->push_back(node_state ? 1.0 : 0.0);
313
314 }
315 new_proba_istates->push_back(new ProbaIState(state_proba.second, state_values));
316
317 }
318
319
320 new IStateGroup(network, nodes, new_proba_istates);
321 }

323 static void setNodeProba(Network * network, Node * node, double value) {
324
325 IStateGroup::removeNode(network, node);
326
327 std::vector<const Node*>* new_nodes = new std::vector<const Node*>();
328
329 new_nodes->push_back(node);
330
331 std::vector<IStateGroup::ProbaIState*>* new_proba_istates = new std::vector<IStateGroup::ProbaIState*>();
332
333 if (value == 0.0 || value == 1.0) {
334 new_proba_istates->push_back(new ProbaIState(1.0, value));
335
336 } else {
337 new_proba_istates->push_back(new ProbaIState(1.0-value, 0.0));
338 new_proba_istates->push_back(new ProbaIState(value, 1.0));
339 }
340
341
342 new IStateGroup(network, new_nodes, new_proba_istates);
343 }
